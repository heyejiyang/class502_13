CREATE OR REPLACE FUNCTION FUNC_AFTERTAX(
	P_SAL NUMBER
)
RETURN NUMBER
IS --선언부
	TAX NUMBER := 0.05;
	SAL NUMBER;
BEGIN --실행부
	SAL := ROUND(P_SAL - P_SAL * TAX);
	RETURN SAL;
END FUNC_AFTERTAX;
